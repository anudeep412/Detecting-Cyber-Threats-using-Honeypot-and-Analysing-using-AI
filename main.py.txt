# Database setup
def init_db():
    """Initialize database with schema"""
    conn = sqlite3.connect('honeypot.db')
    cursor = conn.cursor()

    # Create logs table with better indexing
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS logs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp TEXT NOT NULL,
            source_ip TEXT NOT NULL,
            raw_log TEXT NOT NULL,
            event_type TEXT,
            processed BOOLEAN DEFAULT FALSE,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')

    # Create predictions table with structure
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS predictions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            log_id INTEGER NOT NULL,
            mitre_technique TEXT NOT NULL,
            description TEXT,
            confidence REAL,
            severity TEXT,
            analysis TEXT,
            indicators TEXT,
            risk_level TEXT,
            threat_score REAL,
            recommended_action TEXT,
            false_positive_likelihood TEXT,
            attack_stage TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (log_id) REFERENCES logs (id)
        )
    ''')

    # Create indexes for better performance
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_logs_timestamp ON logs(timestamp)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_logs_source_ip ON logs(source_ip)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_predictions_technique ON predictions(mitre_technique)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_predictions_severity ON predictions(severity)')
    cursor.execute('CREATE INDEX IF NOT EXISTS idx_predictions_risk_level ON predictions(risk_level)')

    conn.commit()
    conn.close()

# Initialize database
init_db()

class ConnectionManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)
        logger.info(f"WebSocket connected. Total connections: {len(self.active_connections)}")

    def disconnect(self, websocket: WebSocket):
        if websocket in self.active_connections:
            self.active_connections.remove(websocket)
        logger.info(f"WebSocket disconnected. Total connections: {len(self.active_connections)}")

    async def broadcast(self, message: str):
        if not self.active_connections:
            return

        disconnected = []
        for connection in self.active_connections:
            try:
                await connection.send_text(message)
            except Exception as e:
                logger.warning(f"Failed to send WebSocket message: {e}")
                disconnected.append(connection)

        # Remove disconnected connections
        for conn in disconnected:
            self.disconnect(conn)

manager = ConnectionManager()

    async def analyze_log_with_gpt4(self, log_entry: str) -> Dict[str, Any]:

def store_log(source_ip: str, raw_log: str, event_type: str = None) -> int:
    """Store log in database with event type"""
    conn = sqlite3.connect('honeypot.db')
    cursor = conn.cursor()

    cursor.execute('''
        INSERT INTO logs (timestamp, source_ip, raw_log, event_type)
        VALUES (?, ?, ?, ?)
    ''', (datetime.now().isoformat(), source_ip, raw_log, event_type))

    log_id = cursor.lastrowid
    conn.commit()
    conn.close()
    return log_id

def get_recent_predictions(limit: int = 10) -> List[Dict[str, Any]]:
    """Get recent predictions with FIXED field mapping"""
    conn = sqlite3.connect('honeypot.db')
    cursor = conn.cursor()

    # Only select columns that actually exist
    cursor.execute('''
        SELECT p.id, p.log_id, p.mitre_technique, p.description, p.confidence,
               p.severity, p.analysis, p.indicators, p.risk_level, p.threat_score,
               p.recommended_action, p.false_positive_likelihood, p.attack_stage,
               p.created_at, l.source_ip, l.timestamp as log_timestamp, l.event_type
        FROM predictions p
        JOIN logs l ON p.log_id = l.id
        ORDER BY p.created_at DESC
        LIMIT ?
    ''', (limit,))

    rows = cursor.fetchall()
    conn.close()

    predictions = []
    for row in rows:
        try:
            indicators = json.loads(row[7]) if row[7] else []
        except json.JSONDecodeError:
            indicators = []

        # Return CONSISTENT structure for frontend
        predictions.append({
            "id": row[0],
            "timestamp": row[15] if row[15] else row[13],  # log_timestamp or created_at
            "technique": row[2],  # Frontend expects this field name
            "mitre_technique": row[2],  # Keep both for compatibility
            "description": row[3] or "Unknown activity",
            "severity": row[5] or "Medium",
            "confidence": row[4] or 70,
            "indicators": indicators,
            "risk_level": row[8] or "MEDIUM",
            "threat_score": row[9] or 50,
            "recommended_action": row[10] or "investigate",
            "false_positive_likelihood": row[11] or "medium",
            "attack_stage": row[12] or "reconnaissance",
            "source_ip": row[14] or "Unknown",
            "event_type": row[16] if len(row) > 16 else "unknown"
        })

    return predictions

def get_dashboard_stats() -> Dict[str, Any]:
    """Get dashboard statistics with FIXED queries"""
    conn = sqlite3.connect('honeypot.db')
    cursor = conn.cursor()

    try:
        # Get total attacks
        cursor.execute('SELECT COUNT(*) FROM predictions')
        total_attacks = cursor.fetchone()[0]

        # Get active threats (last 24 hours)
        cursor.execute('''
            SELECT COUNT(*) FROM predictions
            WHERE created_at > datetime('now', '-1 day')
        ''')
        active_threats = cursor.fetchone()[0]

        # Get unique MITRE techniques
        cursor.execute('SELECT COUNT(DISTINCT mitre_technique) FROM predictions')
        mitre_mapping = cursor.fetchone()[0]

        # Get unique source IPs (last 7 days)
        cursor.execute('''
            SELECT COUNT(DISTINCT l.source_ip) FROM logs l
            JOIN predictions p ON l.id = p.log_id
            WHERE p.created_at > datetime('now', '-7 days')
        ''')
        unique_attackers = cursor.fetchone()[0]

        # Get MITRE technique distribution
        cursor.execute('''
            SELECT mitre_technique, COUNT(*) as count
            FROM predictions
            GROUP BY mitre_technique
            ORDER BY count DESC
            LIMIT 10
        ''')
        technique_counts = cursor.fetchall()

        # Get severity distribution
        cursor.execute('''
            SELECT severity, COUNT(*) as count
            FROM predictions
            GROUP BY severity
            ORDER BY
                CASE severity
                    WHEN 'Critical' THEN 1
                    WHEN 'High' THEN 2
                    WHEN 'Medium' THEN 3
                    WHEN 'Low' THEN 4
                    ELSE 5
                END
        ''')
        severity_counts = cursor.fetchall()

        # Get top attacking IPs with error handling
        cursor.execute('''
            SELECT l.source_ip, COUNT(*) as attack_count,
                   COALESCE(AVG(p.threat_score), 50) as avg_threat_score
            FROM logs l
            JOIN predictions p ON l.id = p.log_id
            WHERE p.created_at > datetime('now', '-7 days')
              AND l.source_ip IS NOT NULL
              AND l.source_ip != ''
            GROUP BY l.source_ip
            ORDER BY attack_count DESC
            LIMIT 5
        ''')
        top_attackers = cursor.fetchall()

    except Exception as e:
        logger.error(f"Database query error: {e}")
        return {
            "totalAttacks": 0,
            "activeThreats": 0,
            "mitreMapping": 0,
            "uniqueAttackers": 0,
            "mitreStats": [],
            "severityStats": [],
            "topAttackers": [],
            "lastUpdate": datetime.now().isoformat(),
            "error": str(e)
        }

    finally:
        conn.close()

    # Process data for frontend with consistent field names
    mitre_stats = []
    for technique, count in technique_counts:
        technique_name = analyzer.mitre_techniques.get(technique, {}).get("name", technique)
        mitre_stats.append({
            "technique": technique,  # Frontend expects this field
            "name": technique_name,
            "count": count,
            "percentage": int((count / max(total_attacks, 1)) * 100)
        })

    severity_stats = [
        {"severity": severity, "count": count}
        for severity, count in severity_counts
    ]

    attacker_stats = [
        {
            "ip": ip,
            "attacks": count,
            "avg_threat_score": round(avg_score, 1) if avg_score else 50
        }
        for ip, count, avg_score in top_attackers
    ]

    return {
        "totalAttacks": total_attacks,
        "activeThreats": active_threats,
        "mitreMapping": mitre_mapping,
        "uniqueAttackers": unique_attackers,
        "mitreStats": mitre_stats,
        "severityStats": severity_stats,
        "topAttackers": attacker_stats,
        "lastUpdate": datetime.now().isoformat()
    }

@app.post("/api/logs")
async def receive_logs(log_data: dict):
    """FIXED: Process logs with pre-computed analysis from Python script"""
    try:
        raw_log = log_data.get('log_data', str(log_data))
        source_ip = log_data.get('source_ip', 'unknown')
        event_type = log_data.get('event_type', 'unknown')

        # CHECK IF ANALYSIS ALREADY PROVIDED BY PYTHON SCRIPT
        pre_analysis = log_data.get('gpt_analysis')

        logger.info(f"Processing {event_type} log from {source_ip}: {raw_log[:100]}...")

        # Store the raw log
        log_id = store_log(source_ip, raw_log, event_type)

        if pre_analysis:
            # USE EXISTING ANALYSIS FROM PYTHON SCRIPT (SAVE MONEY!)
            logger.info("💰 Using pre-computed analysis from Python script")
            analysis = pre_analysis
            analysis_source = "python_script"
        else:
            # FALLBACK: Analyze with backend GPT-4 only if no pre-analysis
            logger.warning("⚠️ No pre-analysis found, analyzing with backend GPT-4")
            analysis = await analyzer.analyze_log_with_gpt4(raw_log)
            analysis_source = "backend"

        # Store the prediction with STANDARDIZED fields
        prediction = store_prediction(log_id, analysis)

        # Add consistent field names for frontend
        prediction.update({
            "source_ip": source_ip,
            "event_type": event_type,
            "analysis_source": analysis_source
        })

        # Broadcast to WebSocket clients with consistent structure
        await manager.broadcast(json.dumps(prediction))

        logger.info(f"✅ Analysis complete: {analysis.get('mitre_technique')} - {analysis.get('severity')} ({analysis.get('confidence', 0)}%) - Source: {analysis_source}")

        return {
            "status": "processed",
            "log_id": log_id,
            "prediction": prediction,
            "analysis_source": analysis_source,
            "cost_saved": analysis_source == "python_script"
        }

    except Exception as e:
        logger.error(f"Error processing log: {e}")
        return {"status": "error", "message": str(e)}

@app.get("/api/cowrie/logs")
async def get_cowrie_logs(limit: int = 50):
    """Get raw Cowrie logs with better error handling - FIXED VERSION"""
    try:
        cowrie_log_path = "/etc/cowrie/var/log/cowrie/cowrie.json"

        if not Path(cowrie_log_path).exists():
            return {"error": "Cowrie log file not found", "logs": []}

        logs = []
        try:
            # READ MUCH MORE LINES to ensure we get recent commands
            with open(cowrie_log_path, 'r') as f:
                lines = f.readlines()

            # Take the last 200 lines instead of just 50 to ensure we get commands
            recent_lines = lines[-200:] if len(lines) > 200 else lines

            # Parse all lines and collect valid entries
            all_logs = []
            for line in recent_lines:
                try:
                    if line.strip():
                        log_entry = json.loads(line.strip())
                        all_logs.append(log_entry)
                except json.JSONDecodeError:
                    continue

            # Sort by timestamp (most recent first) and limit results
            all_logs.sort(key=lambda x: x.get('timestamp', ''), reverse=True)
            logs = all_logs[:limit]

            logger.info(f"Fetched {len(logs)} Cowrie logs from {len(lines)} total lines")

            # Debug: Count event types
            event_counts = {}
            for log in logs:
                event_type = log.get('eventid', 'unknown')
                event_counts[event_type] = event_counts.get(event_type, 0) + 1

            logger.info(f"Event types in response: {event_counts}")

        except Exception as e:
            logger.error(f"Error reading log file: {e}")
            return {"error": f"Error reading log file: {str(e)}", "logs": []}

        return {
            "logs": logs,
            "count": len(logs),
            "file_path": cowrie_log_path,
            "total_lines_read": len(lines) if 'lines' in locals() else 0,
            "event_types": event_counts if 'event_counts' in locals() else {}
        }

    except Exception as e:
        logger.error(f"Error reading Cowrie logs: {e}")
        return {"error": str(e), "logs": []}

            conn.close()
            db_status = "connected"
        except:
            db_status = "disconnected"
            log_count = 0
            prediction_count = 0

        return {
            "status": "healthy",
            "timestamp": datetime.now().isoformat(),
            "openai": {
                "status": openai_status,
                "model": openai_model,
                "api_key_configured": bool(analyzer.openai_api_key)
            },
            "database": {
                "status": db_status,
                "log_count": log_count,
                "prediction_count": prediction_count
            },
            "websocket_connections": len(manager.active_connections),
            "enhanced_features": {
                "pre_computed_analysis": True,
                "cost_optimization": True,
                "consistent_data_structure": True,
                "fixed_database_queries": True,
                "improved_log_reading": True
            }
        }
    except Exception as e:
        return {
            "status": "unhealthy",
            "error": str(e),
            "timestamp": datetime.now().isoformat()
        }
